Find the thirteen adjacent digits in the 1000-digit number that have the greatest product. What is the value of this product?

#include <cmath>
#include <cstdlib>
#include <iostream>
#include <string>

#define max(a, b)                                                              \
  ({                                                                           \
    __typeof__(a) _a = (a);                                                    \
    __typeof__(b) _b = (b);                                                    \
    _a > _b ? _a : _b;                                                         \
  })
